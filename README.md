    options validvarname=upcase;
    libname sd1 "d:/sd1";
    data sd1.have;
      array ys y1-y5 (1,.,5,.,1);
      array xs x1-x5 (1,2,3,4,5);
      do over xs;
        y=ys;
        x=xs;
        keep x y;
        output;
      end;
    run;quit;

    libname sd1 "d:/sd1";
    proc plot data=sd1.have(rename=
       (y=y12345678901234567890123456));
    plot y12345678901234567890123456*x="X"
       /box vaxis=1 to 5 by 1 haxis=1 to 5 by 1;
    run;quit;

    %utl_rbegin;
    parmcards4;
    library(haven);
    source("c:/temp/fn_tosas9.R");
    have<-read_sas("d:/sd1/have.sas7bdat");
    have;
    want<-data.frame(approx(
        have$X
       ,have$Y
       ,n=100
       ,method="linear"));
    want;
    fn_tosas9(dataf=want);
    ;;;;
    %utl_rend;

    libname tmp "c:/temp";
    options ls=64 ps=28;
    proc plot data=tmp.want(rename=
      (y=y12345678901234567890123456));
    plot y12345678901234567890123456*x ='*'
      /box vaxis=1 to 5 by 1 haxis=1 to 5 by 1;
    run;quit;

    proc print data=tmp.want;
    run;quit;
